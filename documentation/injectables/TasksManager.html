  protected addAll(objects: Array<Task>): void {
    const component = this;

              resolve(geoJson);
            }
          }
        })
        .catch(reject);
    });

  public load(objectId: string): Promise<any> {
    if (objectId.includes('carta')) {
      return new Promise((resolve, reject) => {
        this.cuencaVerdeService.getCartaIntencionForEdit(objectId)
          .then(task => {
            task.startdate = task.startdate ? task.startdate.split(' ')[0] : '';
            task.deadline = task.deadline ? task.deadline.split(' ')[0] : '';
            const subType = new Subtype();
            subType.name = 'Carta de intencin';
            task.sub_type = subType;
            task.route = task.id + '_carta';
            this.retrieveInstance(String(task.route), task);
            resolve(task);
          })
          .catch(reject);
      });
    } else {
      return new Promise((resolve, reject) => {
        this.cuencaVerdeService.getTask(objectId)
          .then(task => {
            task.startdate = task.startdate ? task.startdate.split(' ')[0] : '';
            task.deadline = task.deadline ? task.deadline.split(' ')[0] : '';
            task.route = task.id;
            this.retrieveInstance(String(task.route), task);
            resolve(task);
          })
          .catch(reject);
      });
    }
  }

  create(task: Task) {
    return new Promise((resolve, reject) => {
      this.cuencaVerdeService.createTask(task)
        .then((response) => {
          task.id = response.object_id;
          task.sub_type = new Subtype();
          task.sub_type.id = response.sub_type_id;
          task.sub_type.name = response.sub_type_name;
          const key = Number(task.taskType.id) === 5 ? String(task.id) + '_carta' : String(task.id);
          this.pool = this.unShiftPool(key, task, this.pool);
          resolve(task);
        })
        .catch(reject);
    });
  }

  update(task: Task) {
    if (String(task.taskType.id) === '5') {
      return new Promise((resolve, reject) => {
        this.cuencaVerdeService.updateCartaIntencion(task)
          .then(success => {
            this.clearObjects();
            this.loadAllObjects()
              .then(() => {
                const subType = new Subtype();
                subType.name = 'Carta de intencin';
                task.sub_type = subType;
                task.route = task.id + '_carta';
                this.retrieveInstance(String(task.route), task);
                resolve(success);
              });
          })
          .catch(reject);
      });
    } else {
      return new Promise((resolve, reject) => {
        this.cuencaVerdeService.updateTask(task)
          .then(success => {
            this.clearObjects();
            this.loadAllObjects()
              .then(() => {
                this.retrieveInstance(String(task.route), task);
                resolve(success);
              });
          })
          .catch(reject);
      });
    }
  }

  public getObjectForEdit(objectId: string): Promise<BaseObject> {
    return new Promise((resolve, reject) => {
      const task: Task = this.search(objectId);
      if (task && task.user) {
        resolve(task);
      } else {
        this.load(objectId)
          .then(returnedObject => {
            resolve(returnedObject);
          })
          .catch(function (error: any) {
            reject(error);
          });
      }
    });
  }
