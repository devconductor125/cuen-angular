import {Component, OnInit} from '@angular/core';
import {ProceduresManager} from '../../../data/managers/procedures.manager';
import {ActivatedRoute, ParamMap, Router} from '@angular/router';
import {BusMessage, MessagingService} from '../../../data/services/messaging.service';
import {RolesManager} from '../../../data/managers/roles.manager';
import {CreatePoolOfContractsComponent} from '../create-pool-of-contracts/create-pool-of-contracts.component';
import {PoolOfContractsManager} from '../../../data/managers/pool-of-contracts.manager';
import {Procedure} from '../../../data/model/procedure';
import {PoolOfContracts} from '../../../data/model/pool-of-contracts';
import {MappingUtils} from '../../../data/utils/mapping.utils';
import {IOption} from 'ng-select';
import {User} from '../../../data/model/user';
import {AsignarContratistaBolsa} from '../../../data/model/asignarContratistaBolsa';
import {AsignarContratistaProcedure} from '../../../data/model/asignarContratistaProcedure';
import {forEach} from '@angular/router/src/utils/collection';
import {AsignarAccionContratista} from '../../../data/model/asignarAccionContratista';
import {Budget} from '../../../data/model/budget';
import {TasksManager} from '../../../data/managers/tasks.manager';

@Component({
  selector: 'cuenca-edit-pools-of-contracts',
  templateUrl: './edit-pool-of-contracts.component.html',
  styleUrls: ['./edit-pool-of-contracts.component.css']
})
export class EditPoolOfContractsComponent extends CreatePoolOfContractsComponent implements OnInit {

  public contratistas: Array<IOption> = [];

  public objetoAsignacionesCompleto: AsignarContratistaBolsa = new AsignarContratistaBolsa();

  public accionesAsignadas: Array<AsignarContratistaProcedure> = [];

  public characters: Array<IOption> = [];

  public selectedContratista: Array<String> = [];

  public id_pool: string;

  public response: Boolean = false;

  public poolOfContracts: PoolOfContracts = new PoolOfContracts();

  public listTask: Array<any> = [];

  constructor(protected messagingService: MessagingService,
              protected proceduresManager: ProceduresManager,
              protected poolOfContractsManager: PoolOfContractsManager,
              protected router: Router,
              protected tasksManager: TasksManager,
              public rolesManager: RolesManager,
              private activatedRoute: ActivatedRoute) {
    super(messagingService, proceduresManager, poolOfContractsManager, router, rolesManager);
  }

  protected onDataLoaded() {
    this.activatedRoute.paramMap
      .map((params: ParamMap) => +params.get('id') + '')
      .subscribe(idString => {
        const id: number = Number(idString);
        this.id_pool = idString;
        if (id > 0) {
          this.poolOfContractsManager.getObjectForEdit(String(id))
            .then(object => {
              this.poolOfContracts = <PoolOfContracts> object;
              this.mapPoolObjects();
              ////console.log(this.poolOfContracts);

              this.getContratistas();

            });
        } else {
          const link = ['/app/pools-of-contracts'];
          this.router.navigate(link);
        }
      });

    this.getTaskForExecution();
  }

  getContratistas(): void {
    const componente = this;
    this.proceduresManager.getUsers(5)
      .then((users: Array<User>) => {

        if (users.length > 0) {
          componente.contratistas = [];

          users.forEach(function (user: User) {
              componente.contratistas.push(componente.mappingContratistastoSelect(user));
              componente.selectedContratista.push('0');
          });

        }

      });
  }

  mappingContratistastoSelect(user: User): any {
    const objeto: any = {value: String(user.id), label: user.name};
    return objeto;
  }

  changeChekedUnique(idBudget: string, booleanSelected: boolean, id_procedure: string) {
    if (!booleanSelected) {
      const index = this.accionesAsignadas.findIndex(procedimiento => procedimiento.id_procedure === id_procedure);

      const index2 = this.accionesAsignadas[index].data.findIndex(budget => budget.id_budget === String(idBudget));

      const indexElement = this.accionesAsignadas[index].data[index2].id_budget;
      this.selectedContratista[indexElement] = '0';
    } else {
      const index3 = this.procedures.findIndex(process => process.id === id_procedure);
      this.procedures[index3].selected = true;
    }
  }

  changeChekedAll(id_procedure: string, booleanSelected: boolean, procedure: any) {
    const componente = this;

    if (!booleanSelected) {

      const index2 = this.procedures.findIndex(process => process.id === procedure.id);

      this.procedures[index2].budget.forEach(function (process: any) {
        process.selected = false;
      });

      const index = this.accionesAsignadas.findIndex(procedimiento => procedimiento.id_procedure === id_procedure);

      this.accionesAsignadas[index].data.forEach(function (budget: AsignarAccionContratista) {
        componente.selectedContratista[budget.id_budget] = '0';
      });

      this.accionesAsignadas.splice(index, 1);

      ///// console.log(this.accionesAsignadas);

    }
  }

  onSelectedContr(option: IOption, idBudget: string, id_procedure: string) {
    //// console.log(`Contratista IDUSER: ${option.value} - Accin IDBUDGET: ${idBudget}`);

    const index = this.accionesAsignadas.findIndex(procedimiento => procedimiento.id_procedure === id_procedure);

    const objeto: any = {id_procedure: id_procedure, data: []};

    // console.log(index); bien

    if (index === -1) {
      this.accionesAsignadas.push(objeto);
    }

    /// console.log(this.accionesAsignadas); bien

    const index2 = this.accionesAsignadas.findIndex(procedimiento => procedimiento.id_procedure === id_procedure);

    /// console.log(this.accionesAsignadas[index2]);

    const index3 = this.accionesAsignadas[index2].data.findIndex(budget => budget.id_budget === String(idBudget));

    const objeto2: any = {id_user: option.value, id_budget: String(idBudget)};

    if (index3 === -1) {
      this.accionesAsignadas[index2].data.push(objeto2);
    } else {
      this.accionesAsignadas[index2].data.splice(index3, 1);
      this.accionesAsignadas[index2].data.push(objeto2);
    }

    // console.log(this.accionesAsignadas);

    this.objetoAsignacionesCompleto = {
      id_pool: this.id_pool,
      data: this.accionesAsignadas
    };

    ///// console.log(JSON.stringify(this.objetoAsignacionesCompleto));

  }

  private mapPoolObjects() {
    const component = this;
    if (this.poolOfContracts.pool_by_process) {
      this.poolOfContracts.pool_by_process.forEach(function (procedure) {
        component.procedures.forEach(function (componentProcedure: Procedure) {
          if (componentProcedure.id === procedure.process_id) {
            componentProcedure.selected = true;
            componentProcedure.budget.push(MappingUtils.mapPoolOfContractsActionToBudget(procedure));
          }
        });
      });
    }

  }

  ///// POBLAR SELECTS
  public getTaskForExecution(): void {
    const componente = this;
    this.tasksManager.getTaskForExecution()
      .then((response: any) => {
        this.listTask = response;

        this.listTask.forEach((list) => {
          componente.selectedContratista[list.pool_process.budget_id] = list.user_id;
        });

      });

  }

  updatePoolOfContracts(): void {

    const componente = this;

    this.isValidAsi();

   /// console.log(this.response);

    this.setPoolProcedures();
    if (this.isValidPool() && this.response) {

      /////console.log(this.objetoAsignacionesCompleto);

      this.poolOfContractsManager.update(this.poolOfContracts, this.objetoAsignacionesCompleto)
        .then(() => {
            const message = {
              'tipo': 'Registrado',
              'message': 'La bolsa de contratacin ha sido actualizada satisfactoriamente',
              'style': 'alert-success'
            };
            componente.messagingService.publish(new BusMessage('alerta', message));
            const link = ['/app/pools-of-contracts'];
            this.router.navigate(link);
        }, function () {
            const message = {
              'tipo': 'Error',
              'message': 'La bolsa de contratacin no ha sido actualizada',
              'style': 'alert-danger'
            };
            componente.messagingService.publish(new BusMessage('alerta', message));
        });
    }
  }

  protected isValidPool(): boolean {

    ////const objetoSelected = this.procedures.budget.filter(process => process.id === String(this.idMonitor));

    if (!this.poolOfContracts.name || this.poolOfContracts.name.length === 0) {
      const message = {
        'tipo': 'Error: ',
        'message': 'Ingresa el nombre de la bolsa de contratacin',
        'style': 'alert-danger'
      };
      this.messagingService.publish(new BusMessage('alerta', message));
      return false;
    }
    if (this.accionesAsignadas.length === 0 ) {
      const message = {
        'tipo': 'Error: ',
        'message': 'Selecciona una o ms acciones',
        'style': 'alert-danger'
      };
      this.messagingService.publish(new BusMessage('alerta', message));
      return false;
    }



    return true;
  }

  protected isValidAsi() {

    const componente = this;

    this.procedures.forEach((item) => {
        if (item.selected) {

          if (Object.keys(componente.objetoAsignacionesCompleto).length !== 0 && componente.objetoAsignacionesCompleto.constructor !== AsignarContratistaBolsa) {

            const objetoSelected = componente.objetoAsignacionesCompleto
              .data.filter(seleccion => String(seleccion.id_procedure) === String(item.id));

            if (objetoSelected) {

              item.budget.forEach((budget) => {
                if (budget.selected) {

                  if (objetoSelected.length > 0) {

                      const budgetSelected = objetoSelected[0].data.filter((budgetS: any) => String(budgetS.id_budget) === String(budget.id));

                      if (budgetSelected.length === 0) {
                        const message = {
                          'tipo': 'Error: ',
                          'message': 'Faltan datos',
                          'style': 'alert-danger'
                        };
                        this.messagingService.publish(new BusMessage('alerta', message));
                        componente.response = false;
                      } else {
                        componente.response = true;
                      }

                  } else {

                      const message = {
                        'tipo': 'Error: ',
                        'message': 'Faltan datos',
                        'style': 'alert-danger'
                      };
                      this.messagingService.publish(new BusMessage('alerta', message));
                      componente.response = false;

                  }

                }
              });

            } else {

              const message = {
                'tipo': 'Error: ',
                'message': 'Faltan datos',
                'style': 'alert-danger'
              };
              this.messagingService.publish(new BusMessage('alerta', message));
              componente.response = false;

            }

          } else {

            ///console.log('Sin Procedimiento no acciones');

            const message = {
              'tipo': 'Error: ',
              'message': 'Faltan datos',
              'style': 'alert-danger'
            };
            this.messagingService.publish(new BusMessage('alerta', message));
            componente.response = false;
          }

        }
    });

  }
}
